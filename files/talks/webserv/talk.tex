\begin{frame}{Motivation}
  \begin{outline}
    \1 Large Language Model (LLM) Web Agents have been developed and widely used
    \1 WebGPT\cite{nakanoWebGPTBrowserassistedQuestionanswering2022}, Claude Computer Use, ChatGPT Atlas, Step \cite{sodhiStePStackedLLM2024}, LASER \cite{maLASERLLMAgent2024}, WebAgent \cite{gurRealWorldWebAgentPlanning2023}, \dots
  \end{outline}
\end{frame}

\begin{frame}{Motivation}
  \begin{outline}
    \1 These agents have become a primising approach for web automation tasks \cite{zhouWebArenaRealisticWeb2024} and simulating user behaviors \cite{luPromptingNotAll2025}. 
    \1 However, existing Web Agents are mainly built with \textbf{prompting-based} and \textbf{behavior cloning-based} methods.
    \1 RLHF and RLVR have been proven to work in other fields and tasks \cite{ouyangTrainingLanguageModels2022,deepseek-aiDeepSeekR1IncentivizingReasoning2025}
    \1 RL Web Agents is limited by the absense of a full-stack, efficient and effective environment.
  \end{outline}
\end{frame}

\begin{frame}{Web Browsers are NOT RL-Ready}
  \begin{outline}
    \1 Web Browsers are NOT RL-Ready:
      \2 Requires per-site configuration or rely on site features
      \2 Noisy context and action
      \2 Lack of visual hint
      \2 Non-robust action execution
  \end{outline}
\end{frame}

\begin{frame}{Web Servers are NOT RL-Ready}
  \begin{outline}
    \1 Web Servers are usally packed in docker images
    \1 Not possible to run hundsreds \textit{isolated} web server containers to support parallel RL rollouts
  \end{outline}
\end{frame}

\begin{frame}{Existing Work}
  \begin{outline}
    \1 Researchers have argued that existing web environments are not RL-ready \cite{chenScalingAgentLearning2025}
    \1 Existing works are:
      \2 \dots off-policy \cite{qiWebRLTrainingLLM2025}
      \2 \dots not isolated \cite{weiWebAgentR1TrainingWeb2025}
      \2 \dots uses synthetic environments \cite{chenScalingAgentLearning2025}
    \1 The community needs a scalable and efficient training environment 
  \end{outline}
\end{frame}
